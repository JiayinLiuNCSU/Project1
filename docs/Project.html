1.Getting the usual process to work with a given URL and turn the returned info into a nice tibble.

To start any work of data, we need to get the data first. As we want to use API data, it would be nice if the data can be converted to tibble format. To bring in data, we need to build the URL first, and use GET() function to contact the website, then parse the returned data, finally put it into a tibble. Note, it is possible the process failed, thus, it would be safer if we add the option which is ‘fail to access the data’.Based on the information from Google,

After get the returned data, I need to make it to a nice tibble format, however, remember that the first row that returned is not the data or the column name. Thus, we need to removed the first row for the output.I also use drop=False to keep the result as a table and not simplify it into a vector. After this step, we already exclude everything we don’t want so far.The last step is that give the data the column names. And the we should be good to get teh tibble table data by API now.

data_to_tibble <- function(url) {

  returned <- httr::GET(url)

  if (httr::status_code(returned) != 200) {
    stop("FAILED!!!!")
  }

  parsed <- jsonlite::fromJSON(rawToChar(returned$content))

  out <- as_tibble(parsed[-1, , drop = FALSE])

  names(out) <- parsed[1, ]

  out
}

process_numeric_vars <- function(data, num_vars = c("AGEP", "PWGTP")) {
  
  select_num_vars <- c("AGEP", "GASP", "GRPIP", "JWAP", "JWDP", "JWMNP", "PWGTP")
  
  # Check that all requested numeric variables are allowed
  if (!all(num_vars %in% select_num_vars)) {
    stop("Numeric variables NOT AGEP, GASP, GRPIP, JWAP, JWDP, JWMNP, or PWGTP")
  }
  
  # PWGTP should always be returned
  num_vars <- unique(c(num_vars, "PWGTP"))
  
  # One numeric variable other than PWGTP must be returned (Based on the requirement)
  if (length(setdiff(num_vars, "PWGTP")) < 1) {
    stop("One numeric variable other than PWGTP must be returned")
  }
  
  # For JWAP:
  # JWAP: the number changed in every 5 mins interval, for example 006-->12:25am-12:29 am
  
  convert_jwap <- function(x) {
    
    x <- as.numeric(x)
    
    case_when(
      is.na(x) ~ NA,
      TRUE ~ {
        start_min <- x*5
        mid_min <- start_min + 2
        mid_min %% 1440
      }
    )
  }
  
  #For JWDP, since theere are a lot of different pattern of JWDP, I decide manully insert the mapping information, and this can make the whole process easier
 convert_jwdp <- function(x) {
    
    code <- as.character(x)
    
    code <- ifelse(
      code %in% c("0", "00", "000"),
      "0",
      sprintf("%03d", as.numeric(code))
    )
    
    as.numeric(jwdp_map[code])
  }
  
  for (var in num_vars) {
    
    if (var %in% names(data)) {
      
      if (var == "JWAP") {
        data[[var]] <- convert_jwap(data[[var]])
        
      } else if (var == "JWDP") {
        data[[var]] <- convert_jwdp(data[[var]])
        
      } else {
        data[[var]] <- as.numeric(data[[var]])
      }
    }
  }
  
  data
}

# I tried to use ST, DIVISITON, and REGION directly, but the API doesnt allowed me to use these variables directly as predicates, so we use ucgid for API-level subsetting.
process_geo_subset <- function(geography = "ST", geo_values = "24") {
  
  geography <- toupper(geography)
  
  allowed_geography <- c("ST", "DIVISION", "REGION")
  
  if (!(geography %in% allowed_geography)) {
    stop("Geography must be one of: ST, DIVISION, or REGION.")
  }
  
  # If user asks for all values
  if (length(geo_values) == 1 && tolower(geo_values) == "all") {
    
    if (geography == "ST") {
      return(list(ucgid = "040C0US*"))
    }
    
    if (geography == "DIVISION") {
      return(list(ucgid = "030C0US*"))
    }
    
    if (geography == "REGION") {
      return(list(ucgid = "020C0US*"))
    }
  }
  
  geo_values <- as.character(geo_values)
  
  # Add leading zero for one-digit state codes
  if (geography == "ST") {
    
    geo_values <- ifelse(
      nchar(geo_values) == 1,
      paste0("0", geo_values),
      geo_values
    )
    
    ucgid_values <- paste0("040C0US", geo_values)
  }
  
  if (geography == "DIVISION") {
    ucgid_values <- paste0("030C0US", geo_values)
  }
  
  if (geography == "REGION") {
    ucgid_values <- paste0("020C0US", geo_values)
  }
  
  list(
    ucgid = paste(ucgid_values, collapse = ",")
  )
}

get_pums_data <- function(year = 2022,
                          num_vars = c("AGEP", "PWGTP"),
                          cat_vars = "SEX",
                          geography = "ST",
                          geo_values = "24",
                          key) {
  
  # Check year
  year <- check_year(year)
  
  # Check geography
  geography <- toupper(geography)
  
  selected_geography <- c("ST", "DIVISION", "REGION")
  
  if (!(geography %in% selected_geography)) {
    stop("Geography must be one of: ST, DIVISION, or REGION.")
  }
  
  # Check numeric variables
  selected_num_vars <- c(
    "AGEP", "GASP", "GRPIP", "JWAP", "JWDP", "JWMNP", "PWGTP"
  )
  
  if (!all(num_vars %in% selected_num_vars)) {
    stop("Numeric variables must be AGEP, GASP, GRPIP, JWAP, JWDP, JWMNP, or PWGTP.")
  }
  
  # PWGTP must always be returned
  num_vars <- unique(c(num_vars, "PWGTP"))
  
  if (length(setdiff(num_vars, "PWGTP")) < 1) {
    stop("At least one numeric variable other than PWGTP must be returned.")
  }
  
  # Check categorical variables
  selected_cat_vars <- c("FER", "HHL", "SCH", "SCHL", "SEX")
  
  if (!all(cat_vars %in% selected_cat_vars)) {
    stop("Categorical variables must be FER, HHL, SCH, SCHL, or SEX.")
  }
  
  if (length(cat_vars) < 1) {
    stop("At least one categorical variable must be returned.")
  }
  
  # Variables to return
  vars <- unique(c(num_vars, cat_vars, geography))
  
  query_list <- list(
    get = paste(vars, collapse = ","),
    key = as.character(key)
  )
  
  # Add geography subset using ucgid
  geo_query <- process_geo_subset(
    geography = geography,
    geo_values = geo_values
  )
  
  query_list <- c(query_list, geo_query)
  
  # API URL
  api_url <- paste0(
    "https://api.census.gov/data/",
    year,
    "/acs/acs1/pums"
  )
  
  # API call
  res <- httr::GET(
    url = api_url,
    query = query_list
  )
  
  # Read API result
  text <- rawToChar(res$content)
  
  if (!startsWith(text, "[")) {
    stop(text)
  }
  
  parsed <- jsonlite::fromJSON(text)
  
  # Header row
  header_names <- parsed[1, ]
  
  # Make duplicated names unique
  header_names <- make.unique(header_names)
  
  # Data rows
  out <- tibble::as_tibble(
    parsed[-1, , drop = FALSE],
    .name_repair = "unique"
  )
  
  names(out) <- header_names
  
  # If duplicated geography columns were created, keep the original one
  # This handles names like ST and ST.1
  duplicate_geo_cols <- grep(
    paste0("^", geography, "\\."),
    names(out),
    value = TRUE
  )
  
  if (length(duplicate_geo_cols) > 0) {
    out <- dplyr::select(out, -dplyr::all_of(duplicate_geo_cols))
  }
  
  # Process returned variables
  out <- process_numeric_vars(
    data = out,
    num_vars = num_vars
  )
  
  out <- process_categorical_vars(
    data = out,
    cat_vars = cat_vars
  )
  
  out <- process_geography(
    data = out,
    geography = geography
  )
  
  # Add year variable
  out$year <- year
  
  # Add census class
  class(out) <- c("census", class(out))
  
  out
}

get_pums_multi_year <- function(years,
                                num_vars = c("AGEP", "PWGTP"),
                                cat_vars = "SEX",
                                geography = "ST",
                                geo_values = "24",
                                key) {
  
  # Check years like the first function 
  if (!is.numeric(years)) {
    stop("Years should be numeric.")
  }
  
  if (!all(years %in% 2010:2022)) {
    stop("All years between 2010 and 2022.")
  }
  
  # Do not allow 2020
  if (2020 %in% years) {
    stop("Do not include 2020 due to COVID.")
  }
  
  # get the data for each year separately and combine this with the function i wrote before
  out <- purrr::map_dfr(
    years,
    ~ get_pums_data(
      year = .x,
      num_vars = num_vars,
      cat_vars = cat_vars,
      geography = geography,
      geo_values = geo_values,
      key = key
    )
  )
  
  # Add census class
  class(out) <- c("census", class(out))
  
  out
}

get_person_level_records <- function(data) {
  
  # Check PWGTP exists and make it as numeric
  if (!("PWGTP" %in% names(data))) {
    stop("PWGTP mustin the data")
  }
  
  data$PWGTP <- as.numeric(data$PWGTP)
  
  # Check for missing 
  if (any(is.na(data$PWGTP))) {
    stop("PWGTP missing")
  }
  
  if (any(data$PWGTP < 0)) {
    stop("PWGTP not reasonable")
  }
  
  # replicate the data
  person_data <- data %>%
    tidyr::uncount(weights = PWGTP,.remove=FALSE)
  
  person_data
}

summary_census <- function(object,
                           num_vars = NULL,
                           cat_vars = NULL,
                           group_var = NULL,
                           ...) {
  
  # Check PWGTP
  if (!("PWGTP" %in% names(object))) {
    stop("PWGTP must in the data")
  }
  
  object$PWGTP <- as.numeric(object$PWGTP)
  
  # check group variable
  if (!is.null(group_var)) {
    
    if (!(group_var %in% names(object))) {
      stop("group_var not in the data.")
    }
  }
  

    # numeric variables:
  if (is.null(num_vars)) {
    
    num_vars <- names(object)[sapply(object, is.numeric)]
    #remove the variable I don't want to summarize,not useful
    num_vars <- setdiff(num_vars,c("PWGTP", "JWAP", "JWDP"))
  }
  
  # categorical variables:
  if (is.null(cat_vars)) {
    cat_vars <- names(object)[sapply(object, is.factor)]
  }
  
  # Check numeric variables
  if (!all(num_vars %in% names(object))) {
    stop("Some numeric variables not in the data.")
  }
  
  # Check categorical variables
  if (!all(cat_vars %in% names(object))) {
    stop("Some categorical variables  not in the data.")
  }
  
  # Numeric summaries: weighted mean and weighted SD
 numeric_summary <- lapply(num_vars, function(var) {
    
    temp_data <- object %>%
      filter(
        !is.na(.data[[var]]),
        !is.na(PWGTP)
      )
    
    if (!is.null(group_var)) {
      
      temp_data %>%
        group_by(.data[[group_var]]) %>%
        summarise(
          weighted_mean = sum(as.numeric(.data[[var]]) * PWGTP) / sum(PWGTP),
          weighted_sd = sqrt(
            sum((as.numeric(.data[[var]])^2) * PWGTP) / sum(PWGTP) -
              weighted_mean^2
          ),
          .groups = "drop"
        ) %>%
        rename(group = 1) %>%
        mutate(variable = var) %>%
        select(variable, group, weighted_mean, weighted_sd)
      
    } else {
      
      x <- as.numeric(temp_data[[var]])
      w <- temp_data$PWGTP
      
      weighted_mean <- sum(x * w) / sum(w)
      
      weighted_sd <- sqrt(
        sum((x^2) * w) / sum(w) - weighted_mean^2
      )
      
      tibble(
        variable = var,
        weighted_mean = weighted_mean,
        weighted_sd = weighted_sd
      )
    }
  })
  
  names(numeric_summary) <- num_vars
  
  categorical_summary <- lapply(cat_vars, function(var) {
    
    temp_data <- object %>%
      filter(
        !is.na(.data[[var]]),
        !is.na(PWGTP)
      ) %>%
      mutate(level = .data[[var]])
    
    if (!is.null(group_var)) {
      
      temp_data %>%
        group_by(.data[[group_var]], level) %>%
        summarise(
          weighted_count = sum(PWGTP, na.rm = TRUE),
          .groups = "drop"
        ) %>%
        mutate( percent = weighted_count / sum(weighted_count) * 100) %>% 
        rename(group = 1) %>%
        mutate(variable = var) %>%
        select(variable, group, level, weighted_count,percent)
      
    } else {
      
      temp_data %>%
        group_by(level) %>%
        summarise(
          weighted_count = sum(PWGTP, na.rm = TRUE),
          .groups = "drop"
        ) %>%
        mutate(variable = var) %>%
        select(variable, level, weighted_count)
    }
  })
  
  names(categorical_summary) <- cat_vars
  
  out <- list(
    numeric_summary = numeric_summary,
    categorical_summary = categorical_summary
  )
  
  out
}

plot.census <- function(x,cat_var,num_var,...) {
  
  # Check PWGTP 
  if (!("PWGTP" %in% names(x))) {
    stop("PWGTP must be in the data.")
  }
  
  # Check selected variables 
  if (!(cat_var %in% names(x))) {
    stop("The categorical variable not in the data.")
  }
  
  if (!(num_var %in% names(x))) {
    stop("The numeric variable not in the data.")
  }
  
  x$PWGTP <- as.numeric(x$PWGTP)
  x[[num_var]] <- as.numeric(x[[num_var]])
  
  ggplot(
    data = x,
    aes(
      x = get(cat_var),
      y = get(num_var),
      weight = PWGTP
    )
  ) +
    geom_boxplot() +
    labs(
      x = cat_var,
      y = num_var,
      title = paste("Weighted boxplot of", num_var, "by", cat_var)
    )
}
